## Install

```bash
npm install --save @sakit-sa/react-lottie
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from '@sakit-sa/react-lottie'
import '@sakit-sa/react-lottie/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
